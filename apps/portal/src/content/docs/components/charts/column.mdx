The Column chart renders one or more series as vertical bars using the
`CartesianChart` component with a series `type` of
`CartesianSeriesVisualizationType.Column`. Use it to compare a numeric measure
across a small set of discrete categories or time buckets — for example usage
over months, distribution across stages, or counts per bucket.

## Single series

A single column series plotted against a categorical (band) X axis and a linear
Y axis anchored at zero.

<DocsPage.ChartExample
  client:only
  code={`import {
  AxisLocation,
  AxisType,
  CartesianChart,
  CartesianSeriesVisualizationType,
  LegendPosition,
  PopoverServiceProvider,
  ScaleType,
} from "@harness/dashboard-visualization";
import "@harness/dashboard-visualization/style.css";

const series = [
  {
    name: "latency",
    displayName: "Requests",
    color: "#6366f1",
    type: CartesianSeriesVisualizationType.Column,
    data: [
      { x: "0-10ms", y: 42 },
      { x: "10-20ms", y: 18 },
      { x: "20-30ms", y: 9 },
    ],
    stacking: false,
  },
];

const xAxis = {
  type: AxisType.X,
  location: AxisLocation.Bottom,
  scale: ScaleType.Band,
  gridLines: false,
  tickLabels: true,
};

const yAxis = {
  type: AxisType.Y,
  location: AxisLocation.Left,
  scale: ScaleType.Linear,
  gridLines: false,
  tickLabels: true,
  min: 0,
};

render(
  <PopoverServiceProvider>
    <div style={{ width: 480, height: 280 }}>
      <CartesianChart
        series={series}
        xAxisOption={xAxis}
        yAxisOption={yAxis}
        showXAxis
        showYAxis
        legend={LegendPosition.TopRight}
        enableTooltips
        enableCrosshairs
      />
    </div>
  </PopoverServiceProvider>
);`}
/>

## Stacked columns

To stack multiple series on top of one another, set `stacking: true` on **every**
series in the stack. Series share the same band X values and accumulate on the Y
axis.

```tsx
const series: Series<{ x: string; y: number }>[] = [
  {
    name: "compute",
    displayName: "Compute",
    color: "#6366f1",
    type: CartesianSeriesVisualizationType.Column,
    stacking: true,
    data: [
      { x: "Apr", y: 120 },
      { x: "May", y: 90 },
      { x: "Jun", y: 140 },
    ],
  },
  {
    name: "storage",
    displayName: "Storage",
    color: "#22c55e",
    type: CartesianSeriesVisualizationType.Column,
    stacking: true,
    data: [
      { x: "Apr", y: 40 },
      { x: "May", y: 55 },
      { x: "Jun", y: 35 },
    ],
  },
];

<PopoverServiceProvider>
  <CartesianChart
    series={series}
    xAxisOption={xAxis}
    yAxisOption={yAxis}
    showXAxis
    showYAxis
    padding={{ top: 8, left: 0, bottom: 0, right: 16 }}
    legend={LegendPosition.Bottom}
    enableTooltips
  />
</PopoverServiceProvider>;
```

Omitting `stacking` (or setting it to `false`) on multiple series renders them
side-by-side (grouped) across each band instead.

## Per-datum styling

Use `getColor` and `getTooltipTitle` on a series to vary a single column's color
or tooltip header based on its datum — for example hiding zero-value bars.

```tsx
const series: Series<{ x: string; y: number }>[] = [
  {
    name: "usage",
    displayName: "Usage",
    color: "#3b82f6",
    type: CartesianSeriesVisualizationType.Column,
    data: monthlyUsage,
    getColor: (datum) => (datum && datum.y > 0 ? "#3b82f6" : "transparent"),
    getTooltipTitle: (datum) => datum.x,
  },
];
```

## Usage

```tsx
import {
  AxisLocation,
  AxisType,
  CartesianChart,
  CartesianSeriesVisualizationType,
  LegendPosition,
  PopoverServiceProvider,
  ScaleType,
} from "@harness/dashboard-visualization";
import "@harness/dashboard-visualization/style.css";

// A column series is a Series whose `type` is Column. Each data point is an
// { x, y } pair where x is the category label and y is the numeric value.
<PopoverServiceProvider>
  <CartesianChart
    series={[
      {
        name: "series-1",
        color: "#6366f1",
        type: CartesianSeriesVisualizationType.Column,
        data: [{ x: "A", y: 10 }],
      },
    ]}
    xAxisOption={{
      type: AxisType.X,
      location: AxisLocation.Bottom,
      scale: ScaleType.Band,
    }}
    yAxisOption={{
      type: AxisType.Y,
      location: AxisLocation.Left,
      scale: ScaleType.Linear,
      min: 0,
    }}
    legend={LegendPosition.None}
  />
</PopoverServiceProvider>;
```

## Guidelines

<Aside type="tip">
  Keep the Y axis anchored at zero (`min: 0`) for columns — truncating the
  baseline exaggerates differences between bars.
</Aside>

<Aside>
  Use columns for a handful of discrete categories. For dense time-series data
  prefer the Line or Area chart, and for ranking many categories prefer the
  horizontal Bar chart.
</Aside>
