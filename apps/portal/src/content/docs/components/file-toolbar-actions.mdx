The File Toolbar Actions component provides a set of actions for file management, including copying, editing, and downloading files.

import { DocsPage } from "@/components/docs-page";

<DocsPage.ComponentExample
  client:only
  code={`<div className="inline-flex"> 
    <FileToolbarActions 
      onDownloadClick={() => console.log('Download clicked')} 
      onEditClick={() => console.log('Edit clicked')} 
      copyContent="Sample content to copy" 
      showEdit={true} 
    />
  </div>`}
/>

## Usage

```typescript jsx
import { FileToolbarActions } from '@harnessio/ui/components'

return (
  <FileToolbarActions
    onDownloadClick={() => console.log('Download clicked')}
    onEditClick={() => console.log('Edit clicked')}
    copyContent="Sample content to copy"
    showEdit={true}
  />
)
```
