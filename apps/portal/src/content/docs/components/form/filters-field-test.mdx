This page tests the `FilterField` component with various filter types rendered together, mirroring the setup used in the platformUI execution list filters.

import { DocsPage } from "@/components/docs-page";

## All Filter Types Together

Multiple filter types rendered side-by-side, same as how the execution list filter bar works.

<DocsPage.ComponentExample
  client:only
  contentClassName="items-start justify-start"
  code={`() => {
    const [textValue, setTextValue] = React.useState(undefined);
    const [multiSelectValue, setMultiSelectValue] = React.useState(undefined);
    const [dateRangeValue, setDateRangeValue] = React.useState(undefined);
    const [comboBoxValue, setComboBoxValue] = React.useState(undefined);

    const [removedFilters, setRemovedFilters] = React.useState([]);

    const toggleFilter = (key) => {
      setRemovedFilters((prev) =>
        prev.includes(key) ? prev.filter((k) => k !== key) : [...prev, key]
      );
    };

    const statusOptions = [
      { label: 'Success', value: 'Success' },
      { label: 'Failed', value: 'Failed' },
      { label: 'Aborted', value: 'Aborted' },
      { label: 'Running', value: 'Running' },
    ];

    const deploymentTypeOptions = [
      { label: 'Kubernetes', value: 'Kubernetes' },
      { label: 'Native Helm', value: 'NativeHelm' },
      { label: 'ECS', value: 'ECS' },
    ];

    return (
      <Layout.Vertical gapY="md" className="w-full">
        <Layout.Horizontal gapX="sm" className="flex-wrap">
          {removedFilters.length > 0 && (
            <Button variant="secondary" onClick={() => setRemovedFilters([])}>
              Show All Filters
            </Button>
          )}
        </Layout.Horizontal>

        <Layout.Horizontal gapX="sm" className="flex-wrap items-start">
          {!removedFilters.includes('executionId') && (
            <FilterField
              filterOption={{
                type: FilterFieldTypes.Text,
                label: 'Execution ID',
                value: 'executionId',
              }}
              removeFilter={() => toggleFilter('executionId')}
              shouldOpenFilter={false}
              onChange={(val) => setTextValue(val)}
              value={textValue}
            />
          )}

          {!removedFilters.includes('status') && (
            <FilterField
              filterOption={{
                type: FilterFieldTypes.MultiSelect,
                label: 'Status',
                value: 'status',
                filterFieldConfig: {
                  options: statusOptions,
                },
              }}
              removeFilter={() => toggleFilter('status')}
              shouldOpenFilter={false}
              onChange={(val) => setMultiSelectValue(val)}
              value={multiSelectValue}
            />
          )}

          {!removedFilters.includes('deploymentType') && (
            <FilterField
              filterOption={{
                type: FilterFieldTypes.ComboBox,
                label: 'Deployment Type',
                value: 'deploymentType',
                filterFieldConfig: {
                  options: deploymentTypeOptions,
                  placeholder: 'Select deployment type',
                },
              }}
              removeFilter={() => toggleFilter('deploymentType')}
              shouldOpenFilter={false}
              onChange={(val) => setComboBoxValue(val)}
              value={comboBoxValue}
            />
          )}

          {!removedFilters.includes('timeRange') && (
            <FilterField
              filterOption={{
                type: FilterFieldTypes.DateRange,
                label: 'Date Range',
                value: 'timeRange',
              }}
              removeFilter={() => toggleFilter('timeRange')}
              shouldOpenFilter={false}
              onChange={(val) => setDateRangeValue(val)}
              value={dateRangeValue}
            />
          )}
        </Layout.Horizontal>

        <Text color="foreground-2" className="mt-cn-sm">
          Current values: {JSON.stringify({
            executionId: textValue,
            status: multiSelectValue,
            deploymentType: comboBoxValue,
            timeRange: dateRangeValue
              ? { from: dateRangeValue.from?.toISOString?.(), to: dateRangeValue.to?.toISOString?.(), preset: dateRangeValue.preset }
              : undefined,
          }, null, 2)}
        </Text>
      </Layout.Vertical>
    );
}`}
/>

## DateRange Standalone (baseline)

The exact same DateRange filter rendered standalone, confirming it works in isolation.

<DocsPage.ComponentExample
  client:only
  code={`() => {
    const [value, setValue] = React.useState(undefined);
    const [isRemoved, setIsRemoved] = React.useState(false);

    if (isRemoved) {
      return (
        <Button variant="secondary" onClick={() => setIsRemoved(false)}>
          Show Date Range Filter
        </Button>
      );
    }

    return (
      <Layout.Vertical gapY="sm">
        <FilterField
          filterOption={{
            type: FilterFieldTypes.DateRange,
            label: 'Date Range',
            value: 'timeRange',
          }}
          removeFilter={() => setIsRemoved(true)}
          shouldOpenFilter={false}
          onChange={(val) => setValue(val)}
          value={value}
        />
        <Text color="foreground-2">
          Value: {value ? JSON.stringify({ from: value.from?.toISOString?.(), to: value.to?.toISOString?.(), preset: value.preset }) : 'undefined'}
        </Text>
      </Layout.Vertical>
    );
}`}
/>

## DateRange with no filterFieldConfig

DateRange filter configured the same way as in the execution list filter-options.tsx — no filterFieldConfig, relying on default presets.
