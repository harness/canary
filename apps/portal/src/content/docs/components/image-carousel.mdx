The Image Carousel component provides a UI element for displaying a series of images in a carousel format. It includes zoom in and zoom out functionalities.

import { DocsPage } from "@/components/docs-page";

<DocsPage.ComponentExample
  client:only
  code={`() => {
  const [isOpen, setIsOpen] = React.useState(false);
  return <>
    <ImageCarousel 
      isOpen={isOpen} 
      setIsOpen={(value) => setIsOpen(value)} 
      imgEvent={['https://picsum.photos/1240/800', 'https://picsum.photos/seed/picsum/1240/800']} 
      title="Image Carousel" 
      initialSlide={1} 
    />

    <Button onClick={() => setIsOpen(true)}>Open Image Carousel</Button>

</>}`}
/>

## Usage

```typescript jsx
import { ImageCarousel } from '@harnessio/ui/components'

return (
  <ImageCarousel
    isOpen={true}
    setIsOpen={(value) => console.log('Set isOpen:', value)}
    imgEvent={['/path/to/image1.jpg', '/path/to/image2.jpg']}
    title="Image Carousel"
    initialSlide={0}
  />
)
```
