The `LanguageDialog` component allows users to select and save their language preferences.

import { DocsPage } from "@/components/docs-page";

<DocsPage.ComponentExample
  client:only
  code={`<LanguageDialog onChange={() => {}} onSave={() => {}} onCancel={() => {}}><Button>Select Language</Button></LanguageDialog>`}
/>

## Usage

```typescript jsx
import { LanguageDialog } from '@harnessio/ui/components';

//...

return (
  <LanguageDialog
      open={open}
      onOpenChange={setOpen}
      language={language}
      defaultLanguage={LanguageCode.DE}
      onSave={(selectedLanguage: LanguageInterface) => {
        setLanguage(selectedLanguage.code)
        setOpen(false)
      }}
      onChange={(selectedLanguage: LanguageInterface) => setLanguage(selectedLanguage.code)}
      onCancel={() => setOpen(false)}
    >
      <Button onClick={() => setOpen(true)}>Change Language</Button>
  </LanguageDialog>
)
```
