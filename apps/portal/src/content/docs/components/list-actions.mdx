The List Actions component provides a UI element for displaying actions that can be performed on a list of items. It includes dropdown menus for selecting actions.

import { DocsPage } from "@/components/docs-page";

<DocsPage.ComponentExample
  client:only
  code={`<ListActions.Root>
  <ListActions.Left>
    <ListActions.Dropdown 
      title="Actions" 
      items={[
        { name: 'Edit', value: 'edit' },
        { name: 'Delete', value: 'delete' }
      ]}
      onChange={(value) => console.log('Selected:', value)}
    />
  </ListActions.Left>
  <ListActions.Right>
    <button className="btn">Add Item</button>
  </ListActions.Right>
</ListActions.Root>`}
/>

## Usage

```typescript jsx
import { ListActions } from '@harnessio/ui/components'

return (
  <ListActions.Root>
    <ListActions.Left>
      <ListActions.Dropdown
        title="Actions"
        items={[
          { name: 'Edit', value: 'edit' },
          { name: 'Delete', value: 'delete' }
        ]}
        onChange={(value) => console.log('Selected:', value)}
      />
    </ListActions.Left>
    <ListActions.Right>
      <button className="btn">Add Item</button>
    </ListActions.Right>
  </ListActions.Root>
)
```
