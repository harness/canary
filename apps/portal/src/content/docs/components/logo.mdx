The `Logo` component renders brand icons from the `simple-icons` package.

import { DocsPage } from "../../../components/docs-page";
import { Logo, LogoNameMap, Table } from "@harnessio/ui/components";

<DocsPage.ComponentExample
  client:only
  code={`<div className="grid place-items-center">
    <Logo name="kubernetes" size={48} />
</div>`}
/>

## Usage

```typescript jsx
import {  Logo } from '@harnessio/ui/components'

//...

return <Logo name="kuvernetes" size={48} />
```

## API Reference

```typescript jsx
<Logo
  name="logo-name"     // name of the logo
  size={24}            // [OPTIONAL] size of the logo in pixels
  className="my-class" // [OPTIONAL] custom class name
  original             // [OPTIONAL] if false, the logo will use the current text color,
                       // otherwise will use its brand-specific color
/>
```
