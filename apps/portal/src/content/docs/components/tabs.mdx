## Usage

```typescript jsx
import { Tabs, TabsList, TabsTrigger, TabsContent } from '@harnessio/ui/components'

return (
  <Tabs defaultValue="tab1">
    <TabsList variant="default">
      <TabsTrigger variant="default" value="tab1">Tab 1</TabsTrigger>
      <TabsTrigger variant="default" value="tab2">Tab 2</TabsTrigger>
    </TabsList>
    <TabsContent value="tab1">Content for Tab 1</TabsContent>
    <TabsContent value="tab2">Content for Tab 2</TabsContent>
  </Tabs>
)
```

## Anatomy

The Tabs component can be used to create tabbed navigation interfaces with various styles and sizes.

```typescript jsx
<Tabs defaultValue="tab1">
  <TabsList variant="default">
    <TabsTrigger variant="default" value="tab1">Tab 1</TabsTrigger>
    <TabsTrigger variant="default" value="tab2">Tab 2</TabsTrigger>
  </TabsList>
  <TabsContent value="tab1">Content for Tab 1</TabsContent>
  <TabsContent value="tab2">Content for Tab 2</TabsContent>
</Tabs>
```
