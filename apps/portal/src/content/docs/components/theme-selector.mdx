The `ThemeDialog` component allows users to adjust appearance settings such as mode, contrast, color adjustment, and accent color.

import { DocsPage } from "@/components/docs-page";

<DocsPage.ComponentExample

/>

## Usage

```typescript jsx
import { ThemeDialog } from '@harnessio/ui/components';

//...

return (
  <ThemeDialog
    open={open}
    onOpenChange={setOpen}
    theme={theme}
    defaultTheme={{
      mode: Mode.Light,
      contrast: Contrast.HighContrast,
      colorAdjustment: ColorAdjustment.Deuteranopia,
      accentColor: AccentColor.Orange
    }}
    onSave={(appliedTheme: ThemeInterface) => {
      setTheme(appliedTheme)
      setOpen(false)
    }}
    onChange={(updatedTheme: ThemeInterface) => setTheme(updatedTheme)}
    onCancel={() => setOpen(false)}
  >
    <Button onClick={() => setOpen(true)}>Appearance Settings</Button>
  </ThemeDialog>
)
```
