The `ThemeDialog` component allows users to adjust appearance settings such as mode, contrast, color adjustment, and accent color.

## Usage

```typescript jsx
import { ThemeDialog } from '@harnessio/ui/components';

function App() {

  const handleOpenChange = (open: boolean) => {
    console.log(open);
  };

  const handleChange = (theme: Partial<ThemeInterface>) => {
    console.log(theme);
  };

  const handleSave = (theme: ThemeInterface) => {
    console.log(theme);
  };

  const handleCancel = () => {
    console.log('Cancelled');
  };

  return (
    <div>
      <ThemeDialog
        open={true}
        onOpenChange={handleOpenChange}
        onChange={handleChange}
        onSave={handleSave}
        onCancel={handleCancel}
      >
        <button>Open Theme Dialog</button>
      </ThemeDialog>
    </div>
  );
}

export default App;
```
