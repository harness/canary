    const [formData, setFormData] = React.useState({
      name: '',
      email: '',
      role: 'user',
      notifications: true,
      theme: 'light'
    });

    return (
      <Card.Root className="max-w-lg">
        <Card.Title>User Settings</Card.Title>
        <Card.Content className="space-y-cn-lg">
          <Text color="foreground-3">Update your account preferences</Text>

          <TextInput
            label="Full Name"
            placeholder="Enter your name"
            value={formData.name}
            onChange={(e) => setFormData({...formData, name: e.target.value})}
          />

          <TextInput
            label="Email"
            placeholder="Enter your email"
            value={formData.email}
            onChange={(e) => setFormData({...formData, email: e.target.value})}
            caption="We'll never share your email"
          />

          <Select
            label="Role"
            options={[
              { label: 'User', value: 'user' },
              { label: 'Admin', value: 'admin' },
              { label: 'Moderator', value: 'moderator' },
            ]}
            value={formData.role}
            onChange={(value) => setFormData({...formData, role: value})}
          />

          <Switch
            id="notifications"
            label="Email Notifications"
            caption="Receive updates about your account"
            checked={formData.notifications}
            onCheckedChange={(checked) => setFormData({...formData, notifications: checked})}
          />

          <Radio.Root
            value={formData.theme}
            onValueChange={(value) => setFormData({...formData, theme: value})}
            label="Preferred Theme"
          >
            <Radio.Item value="light" label="Light" />
            <Radio.Item value="dark" label="Dark" />

      </Card.Root>
