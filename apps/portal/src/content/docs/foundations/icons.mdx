This component allows you to render various icons with customizable properties such as size, color theme, and name.

## Further reading

- Read [about how-to guides](https://diataxis.fr/how-to-guides/) in the Di√°taxis framework

import { DocsPage } from "@/components/docs-page";
import { Icon, IconNameMap } from "@harnessio/ui/components";

<DocsPage.ComponentExample
  client:only
  code={`
    <div className="flex flex-wrap gap-4 mt-4">
      {Object.keys(IconNameMap).map(name => (
        <div key={name} className="flex items-center gap-4">
          <div className="group relative">
            <div className="absolute bottom-full left-1/2 transform -translate-x-1/2 mb-2 hidden group-hover:block bg-cn-background-3 text-cn-foreground-1 text-xs rounded py-1 px-2">
              {name}
            </div>
            <div className="p-2 rounded-md border border-1">
              <Icon name={name as keyof typeof IconNameMap} size={24} />
            </div>
          </div>
        </div>
      ))}
    </div>
  `}
/>

## Usage

import { Icon } from '@harnessio/ui/components'
